SELECT NUME, PRENUME
FROM Student
WHERE NR_MATR IN (2,4,6,8);

SELECT NUME
FROM Student
WHERE DATA_N BETWEEN '01-May-2002' AND '02-May-2003' AND GEN = 'M';

FROM Student
WHERE PRENUME LIKE 'A%' OR NR_MATR BETWEEN 2 AND 12;
